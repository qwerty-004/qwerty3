%{
#include <stdio.h>
#include <stdlib.h>

int yylex();
int yyerror(const char *s);
%}

%token ID

%%

E  : T E1        { printf("E -> T E'\n"); } ;

E1 : '+' T E1    { printf("E' -> + T E'\n"); }
   |             { printf("E' -> epsilon\n"); } ;

T  : F           { printf("T -> F\n"); } ;

F  : '(' E ')'   { printf("F -> ( E )\n"); }
   | ID          { printf("F -> id\n"); } ;

%%

int main() {
    printf("Enter expression:\n");

    if (yyparse() == 0)
        printf("Parsing Successful!\n");
    else
        printf("Parsing Failed\n");

    return 0;
}
int yyerror(const char *s) {
    printf("Parsing Failed\n");
    return 0;
}


%{
#include "y.tab.h"
%}

%%

[a-zA-Z]+   { return ID; }
[ \t\n]     ;      
"+"         { return '+'; }
"("         { return '('; }
")"         { return ')'; }

%%

int yywrap() {
    return 1;
}

1. Save this file as parser.y
2. Save Lex program as lexer.l

5. Compile:
   gcc y.tab.c lex.yy.c -o parser -ll

6. Run:
   ./parser

7. Sample Input:
   a+b

8. Sample Output:
   F -> id

   E -> T E'
   Parsing Successful!
