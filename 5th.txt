%{
#include <stdio.h>
#include <stdlib.h>

int yylex();
int yyerror(const char *s);
%}

%token ID

%%

E  : T E1        { printf("E -> T E'\n"); } ;

E1 : '+' T E1    { printf("E' -> + T E'\n"); }
   |             { printf("E' -> epsilon\n"); } ;

T  : F           { printf("T -> F\n"); } ;

F  : '(' E ')'   { printf("F -> ( E )\n"); }
   | ID          { printf("F -> id\n"); } ;

%%

int main() {
    printf("Enter expression:\n");

    if (yyparse() == 0)
        printf("Parsing Successful!\n");
    else
        printf("Parsing Failed\n");

    return 0;
}
int yyerror(const char *s) {
    printf("Parsing Failed\n");
    return 0;
}
