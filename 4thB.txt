#include <stdio.h>
#include <string.h>

char input[100];
int i = 0;

// Function declarations
void S();
void L();
void Lprime();

// S → (L) | a
void S() {
    if(input[i] == 'a') {
        i++;
    }
    else if(input[i] == '(') {
        i++;
        L();
        if(input[i] == ')')
            i++;
        else
            printf("Error: Missing )\n");
    }
    else {
        printf("Error in S\n");
    }
}

// L → S L'
void L() {
    S();
    Lprime();
}

// L' → , S L' | ε
void Lprime() {
    if(input[i] == ',') {
        i++;
        S();
        Lprime();
    }
}

int main() {
    printf("Enter input string: ");
    scanf("%s", input);

    S();

    if(input[i] == '\0') {
        printf("String Accepted\n");
    } else {
        printf("String Rejected\n");
    }

    return 0;
}
